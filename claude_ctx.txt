```rust
/// Returns a handle like CreateFrameBuffer. Pixels are zeroed (transparent).
pub(crate) unsafe fn create_owned_framebuffer(width: i32, height: i32) -> i32 {
    let size = (width * height * 4) as usize;
    let mut owned: Vec<u8> = vec![0u8; size];
    let ptr = owned.as_mut_ptr();
    // SAFETY: owned is stored in fb._owned_pixels (same struct), never reallocated.
    let pixels: &'static mut [u8] = std::slice::from_raw_parts_mut(ptr, size);
    let fb = FrameBuffer {
        pixels,
        _owned_pixels: Some(owned),
        w: width,
        h: height,
        antialias: true,
        ctm: Transform::identity(),
        clip_mask: None,
        gstate_stack: Vec::new(),
    };
    let mut map = FB_MAP.write();
    let id = NEXT_FB_ID;
    NEXT_FB_ID += 1;
    map.insert(id, Mutex::new(fb));
    id

        let Some(lock) = map.get(&src_handle) else {
            return;
        };
        let src = lock.lock();
        (src.pixels.to_vec(), src.w, src.h)
    };

    if src_w <= 0 || src_h <= 0 {
        return;
    }

    with_fb(dst_handle, |dst| {
        // Respect dst's clip_mask (take/restore to avoid clone).
        let clip_mask = dst.clip_mask.take();
        if let Some(src_pm) =
            PixmapMut::from_bytes(&mut src_data, src_w as u32, src_h as u32)
        {
            if let Some(mut dst_pm) = dst.pixmap_mut() {
                dst_pm.draw_pixmap(
                    x,
                    y,
                    src_pm.as_ref(),
                    &tiny_skia::PixmapPaint {
                        opacity: alpha,
                        blend_mode: BlendMode::SourceOver,
                        quality: tiny_skia::FilterQuality::Nearest,
                    },
                    Transform::identity(),
                    clip_mask.as_ref(),
                );
            }
        }
        dst.clip_mask = clip_mask;
    });
}
```

```python
def create_owned(cls, width: int, height: int) -> "FrameBuffer":
    """Create a FrameBuffer with its own pixel allocation (no Python buffer needed)."""
    lib = cls._ensure_lib_loaded()
    handle = lib.CreateOwnedFB(width, height)

    fb._lib = lib
    fb._width = width
    fb._height = height
```

+    x: i32,
+    y: i32,
+    alpha: f32,
+) {
+    framebuffer::composite_framebuffer(dst_handle, src_handle, x, y, alpha)
+}
+
Все виглядає коректно. Підсумок змін:

Rust:

CreateOwnedFB(w, h) — власний буфер для view
ClearFB(handle) — очищення до прозорого
CompositeFB(dst, src, x, y, alpha) — blit з урахуванням dst clip_mask
Python:

FrameBuffer.create_owned(w, h), .clear(), .composite_into(dst, x, y, alpha)
Root (parent_layer=None): стара поведінка без змін
Subview path: власний layer → render → composite
Виграш:

Subview з corner_radius=0 (типово): нуль Mask операцій, нуль gstate push/pop на межі view
corner_radius>0: Mask розміром view (не екрану)
Dirty caching: dirty views рендеряться, чисті — тільки composite (дешевий blit)
